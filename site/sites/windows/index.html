$users = Get-ADUser -Filter *
foreach ($user in $users) {
    # Check last login
    $lastLogon = $user.LastLogonDate
    if ($lastLogon -lt (Get-Date).AddDays(-90)) {
        Disable-ADAccount -Identity $user.SamAccountName
    }
}
